Tên Hàm Công dụng Cú pháp mẫu (Copy dùng ngay)
--- Nhóm Xử lý Chuỗi (Text) ---
LEFT Lấy ký tự bên trái =LEFT(Ô_chứa_chữ, Số_ký_tự)
RIGHT Lấy ký tự bên phải =RIGHT(Ô_chứa_chữ, Số_ký_tự)
MID Lấy ký tự ở giữa =MID(Ô_chứa_chữ, Vị_trí_bắt_đầu, Số_ký_tự)
TEXT Đổi định dạng số/ngày =TEXT(Ô_ngày_tháng, "dd/mm/yyyy")
--- Nhóm Ngày Tháng (Date & Time) ---
TODAY Lấy ngày hôm nay =TODAY()
DATEDIF Tính khoảng cách 2 ngày =DATEDIF(Ngày_đầu, Ngày_cuối, "d")
EOMONTH Ngày cuối cùng của tháng =EOMONTH(Ngày_bắt_đầu, 0)
--- Nhóm Tìm Kiếm & Tham Chiếu ---
HLOOKUP Dò tìm theo hàng ngang =HLOOKUP(Giá_trị, Bảng, Hàng_số, 0)
MATCH Tìm vị trí của giá trị =MATCH(Giá_trị, Vùng_tìm, 0)
INDEX Lấy giá trị tại tọa độ =INDEX(Vùng_dữ_liệu, Hàng_số, Cột_số)
